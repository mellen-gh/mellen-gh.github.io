<script type="module">
  import { o3 } from "https://cdn.skypack.dev/goss_proto@1.0.0";
 
  function getPrototypeChain(obj) {
    let prot = Object.getPrototypeOf(obj);
    let name = obj.name;
    let cn = 0;
 
    while (prot) {
      name +=
              `\n` +
              `       `.repeat(cn) +
              `|` +
              `\n` +
              `       `.repeat(cn) +
              `--> ` +
              `${prot.name}`;
      prot = Object.getPrototypeOf(prot);
      cn += 1;
    }
    return name;
  }
 
  alert("Chain of prototypes:\n" + getPrototypeChain(o3));
  console.log(o3);
</script>
